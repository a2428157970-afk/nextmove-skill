**Goal:** Add a validated, transport-neutral request contract to career analysis without changing the Skill Core or fixed workflow.

**Tech Stack:** Python dataclasses and unittest; existing `skill.schemas` and `skill.utils.to_dict`; no new dependencies.

- Keep `application -> public skill API -> Skill Core`; `skill/` must not import `application/`.
- Preserve call order: `analyze_resume`, `improve_resume`, `match_job`, `career_advice`.
- Normalize optional `job_description=None` to `""`; never skip `match_job` or generate a job description.

```python
# application/schemas/request.py
from dataclasses import dataclass
from skill.schemas.resume import ResumeProfile

@dataclass(slots=True)
class CareerAnalysisRequest:
    resume: ResumeProfile | str
    job_description: str | None = None

    def validate(self) -> None:
        if not isinstance(self.resume, (ResumeProfile, str)):
            raise ValueError("resume must be a ResumeProfile or str")
        if isinstance(self.resume, str) and not self.resume.strip():
            raise ValueError("resume must not be empty")
        if not isinstance(self.job_description, (str, type(None))):
            raise ValueError("job_description must be a str or None")

    def normalized_job_description(self) -> str:
        return "" if self.job_description is None else self.job_description
```

```python
def analyze(self, request: CareerAnalysisRequest) -> ApplicationResponse:
    try:
        request.validate()
    except ValueError as exc:
        message = str(exc)
        return ApplicationResponse(success=False, error_code="APPLICATION_VALIDATION_ERROR", failed_step="request_validation", message=message, error=SkillError(code="APPLICATION_VALIDATION_ERROR", message=message))
    return self.workflow.run(request.resume, request.normalized_job_description())
```

```python
def test_application_uses_only_public_skill_interfaces(self):
    application_root = Path(__file__).resolve().parents[1] / "application"
    forbidden = []
    for source_file in application_root.rglob("*.py"):
        for number, line in enumerate(source_file.read_text(encoding="utf-8").splitlines(), 1):
            item = line.strip()
            if item.startswith("from skill.") and not item.startswith(("from skill.schemas", "from skill.utils")):

    self.assertEqual(forbidden, [])
```
